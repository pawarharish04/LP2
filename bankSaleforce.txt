Bank Account System using Salesforce Visualforce (VFP)
Aim
Develop a menu-driven Bank Account System in Salesforce using Custom Object, Apex Controller, and Visualforce Page (VFP).
Technologies Used
• Salesforce
• Apex
• Visualforce
• SOQL
• DML Operations
Project Architecture
Visualforce Page (UI)
        ↓
Apex Controller
        ↓
Customer__c Object
        ↓
Salesforce Database
Step 1 - Create Salesforce Developer Account
Visit https://developer.salesforce.com/signup and create a developer account.
Step 2 - Create Custom Object
Go to:
Setup → Object Manager → Create → Custom Object

Fill:
Label: Customer
Plural Label: Customers
Object Name: Customer
Record Name: Customer Name
Step 3 - Create Fields
1. Email Field
API Name: Email__c

2. Birth Date Field
API Name: Birth_Date__c

3. Department Field
API Name: Department__c
Step 4 - Open Developer Console
Gear Icon → Developer Console
Step 5 - Create Apex Controller
File → New → Apex Class
Class Name: BankVFController
Apex Controller Code
public class BankVFController {

    public String customerName {get; set;}
    public String email {get; set;}
    public Date birthDate {get; set;}
    public String department {get; set;}

    public List<Customer__c> customerList {get; set;}

    public BankVFController() {
        loadCustomers();
    }

    public void loadCustomers() {

        customerList = [
            SELECT Id,
                   Name,
                   Email__c,
                   Birth_Date__c,
                   Department__c
            FROM Customer__c
            ORDER BY CreatedDate DESC
        ];
    }

    public void addCustomer() {

        Customer__c cust = new Customer__c();

        cust.Name = customerName;
        cust.Email__c = email;
        cust.Birth_Date__c = birthDate;
        cust.Department__c = department;

        insert cust;

        loadCustomers();
    }

    public void deleteCustomer() {

        String custId =
            ApexPages.currentPage()
            .getParameters()
            .get('custId');

        Customer__c cust = [
            SELECT Id
            FROM Customer__c
            WHERE Id = :custId
            LIMIT 1
        ];

        delete cust;

        loadCustomers();
    }
}
Step 6 - Create Visualforce Page
File → New → Visualforce Page
Page Name: BankVFPage
Visualforce Page Code
<apex:page controller="BankVFController">

    <apex:form>

        <apex:pageBlock title="Bank Account System">

            <apex:pageMessages />

            <apex:pageBlockSection columns="2">

                <apex:inputText
                    value="{!customerName}"
                    label="Customer Name" />

                <apex:inputText
                    value="{!email}"
                    label="Email" />

                <apex:inputText
                    value="{!birthDate}"
                    label="Birth Date" />

                <apex:inputText
                    value="{!department}"
                    label="Department" />

            </apex:pageBlockSection>

            <apex:commandButton
                value="Add Customer"
                action="{!addCustomer}"
                rerender="customerTable" />

        </apex:pageBlock>

        <apex:pageBlock
            title="Customer Records"
            id="customerTable">

            <apex:pageBlockTable
                value="{!customerList}"
                var="c">

                <apex:column value="{!c.Name}" />
                <apex:column value="{!c.Email__c}" />
                <apex:column value="{!c.Birth_Date__c}" />
                <apex:column value="{!c.Department__c}" />

                <apex:column headerValue="Action">

                    <apex:commandButton
                        value="Delete"
                        action="{!deleteCustomer}">

                        <apex:param
                            name="custId"
                            value="{!c.Id}" />

                    </apex:commandButton>

                </apex:column>

            </apex:pageBlockTable>

        </apex:pageBlock>

    </apex:form>

</apex:page>
